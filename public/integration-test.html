GraphCycle Integration Example

    
Example code for integrating with GraphCycle:

    
// In GraphCycle application
const FUSIONFORMS_URL = 'http://dev-codex.idoxgroup.local/fusionforms';

function openInFusionForms(formData) {
  // Use localStorage for reliability
  localStorage.setItem('fusionforms_import', JSON.stringify(formData));
  window.open(`${FUSIONFORMS_URL}?import=local`, '_blank');
  
  // Clean up after 10 seconds
  setTimeout(() => {
    localStorage.removeItem('fusionforms_import');
  }, 10000);
}
